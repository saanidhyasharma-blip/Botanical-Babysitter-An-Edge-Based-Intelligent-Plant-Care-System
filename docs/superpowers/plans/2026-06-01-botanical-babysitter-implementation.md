# Botanical Babysitter Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a full-stack IoT platform for smart plant monitoring, including a Next.js dashboard, Supabase backend, and ESP32 firmware architecture.

**Architecture:** ESP32 edge device reads sensors and posts to Supabase. Next.js dashboard subscribes to Supabase real-time updates. Device commands are issued via a polling table in Supabase.

**Tech Stack:** Next.js 15 (App Router), TypeScript, Tailwind CSS v4, shadcn/ui, Supabase (PostgreSQL, Auth, Realtime), Zustand, TanStack Query, Recharts, Framer Motion, Arduino C++ (ESP32).

```sql
-- supabase/migrations/00000000000000_initial_schema.sql

-- Enable pgcrypto for UUID generation
CREATE EXTENSION IF NOT EXISTS pgcrypto;

-- User profiles
CREATE TABLE public.profiles (
  id uuid PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
  email text NOT NULL,
  name text,
  avatar_url text,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

-- Registered devices
CREATE TABLE public.devices (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id uuid REFERENCES public.profiles(id) ON DELETE CASCADE,
  device_name text NOT NULL DEFAULT 'My Plant',
  device_uid text UNIQUE NOT NULL,
  firmware_version text,
  status text DEFAULT 'offline' CHECK (status IN ('online','offline')),
  last_seen timestamptz,
  created_at timestamptz DEFAULT now()
);

-- Time-series sensor data
CREATE TABLE public.sensor_readings (
  id bigint GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  device_id uuid REFERENCES public.devices(id) ON DELETE CASCADE,
  temperature real,
  humidity real,
  moisture real,
  pump_status boolean DEFAULT false,
  mode text DEFAULT 'manual' CHECK (mode IN ('auto','manual')),
  created_at timestamptz DEFAULT now()
);

-- Alerts generated by the system
CREATE TABLE public.alerts (
  id bigint GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  device_id uuid REFERENCES public.devices(id) ON DELETE CASCADE,
  severity text NOT NULL CHECK (severity IN ('critical','warning','info')),
  title text NOT NULL,
  message text,
  read boolean DEFAULT false,
  created_at timestamptz DEFAULT now()
);

-- Activity events
CREATE TABLE public.events (
  id bigint GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  device_id uuid REFERENCES public.devices(id) ON DELETE CASCADE,
  event_type text NOT NULL,
  message text,
  created_at timestamptz DEFAULT now()
);

-- Device Commands (Polling Queue)
CREATE TABLE public.device_commands (
  id bigint GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  device_id uuid REFERENCES public.devices(id) ON DELETE CASCADE,
  command text NOT NULL,
  executed boolean DEFAULT false,
  created_at timestamptz DEFAULT now()
);

-- Indexes
CREATE INDEX idx_sensor_readings_device_time ON public.sensor_readings(device_id, created_at DESC);
CREATE INDEX idx_alerts_device_time ON public.alerts(device_id, created_at DESC);
CREATE INDEX idx_events_device_time ON public.events(device_id, created_at DESC);
CREATE INDEX idx_commands_device_executed ON public.device_commands(device_id, executed);

-- Realtime
alter publication supabase_realtime add table public.sensor_readings;
alter publication supabase_realtime add table public.alerts;
alter publication supabase_realtime add table public.events;

-- RLS (Row Level Security) Setup
ALTER TABLE public.profiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.devices ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.sensor_readings ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.alerts ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.events ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.device_commands ENABLE ROW LEVEL SECURITY;

-- Profiles Policies
CREATE POLICY "Users can view own profile" ON public.profiles FOR SELECT USING (auth.uid() = id);
CREATE POLICY "Users can update own profile" ON public.profiles FOR UPDATE USING (auth.uid() = id);

-- Devices Policies
CREATE POLICY "Users can view own devices" ON public.devices FOR SELECT USING (auth.uid() = user_id);
CREATE POLICY "Users can insert own devices" ON public.devices FOR INSERT WITH CHECK (auth.uid() = user_id);
CREATE POLICY "Users can update own devices" ON public.devices FOR UPDATE USING (auth.uid() = user_id);
CREATE POLICY "Users can delete own devices" ON public.devices FOR DELETE USING (auth.uid() = user_id);

-- Sensor Readings Policies
CREATE POLICY "Users can view own sensor readings" ON public.sensor_readings FOR SELECT USING (
  EXISTS (SELECT 1 FROM public.devices WHERE devices.id = sensor_readings.device_id AND devices.user_id = auth.uid())
);
-- Devices insert via Service Role Key (no policy needed for service role)

-- Alerts Policies
CREATE POLICY "Users can view own alerts" ON public.alerts FOR SELECT USING (
  EXISTS (SELECT 1 FROM public.devices WHERE devices.id = alerts.device_id AND devices.user_id = auth.uid())
);
CREATE POLICY "Users can update own alerts" ON public.alerts FOR UPDATE USING (
  EXISTS (SELECT 1 FROM public.devices WHERE devices.id = alerts.device_id AND devices.user_id = auth.uid())
);

-- Events Policies
CREATE POLICY "Users can view own events" ON public.events FOR SELECT USING (
  EXISTS (SELECT 1 FROM public.devices WHERE devices.id = events.device_id AND devices.user_id = auth.uid())
);

-- Commands Policies
CREATE POLICY "Users can view own commands" ON public.device_commands FOR SELECT USING (
  EXISTS (SELECT 1 FROM public.devices WHERE devices.id = device_commands.device_id AND devices.user_id = auth.uid())
);
CREATE POLICY "Users can insert own commands" ON public.device_commands FOR INSERT WITH CHECK (
  EXISTS (SELECT 1 FROM public.devices WHERE devices.id = device_commands.device_id AND devices.user_id = auth.uid())
);
```

```cpp
// firmware/esp32_botanical/esp32_botanical.ino
#include <WiFi.h>
#include <HTTPClient.h>
#include <ArduinoJson.h>
#include "DHT.h"

// --- Config ---
const char* ssid = "YOUR_WIFI_SSID";
const char* password = "YOUR_WIFI_PASSWORD";
const char* supabaseUrl = "https://YOUR_PROJECT.supabase.co/rest/v1";
const char* supabaseKey = "YOUR_SERVICE_ROLE_KEY"; // Service role key needed to bypass RLS for inserts
const char* deviceId = "YOUR_DEVICE_UUID_FROM_DB";

#define DHTPIN 4
#define DHTTYPE DHT11
#define MOISTURE_PIN 34
#define RELAY_PIN 5

DHT dht(DHTPIN, DHTTYPE);

bool pumpStatus = false;
String mode = "manual";
int moistureThreshold = 30;
unsigned long lastReadingTime = 0;
unsigned long lastPollTime = 0;
const long readingInterval = 30000; // 30s
const long pollInterval = 5000;     // 5s

void setup() {
  Serial.begin(115200);
  
  pinMode(RELAY_PIN, OUTPUT);
  digitalWrite(RELAY_PIN, LOW);
  
  dht.begin();
  
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWiFi connected");
}

void loop() {
  unsigned long currentMillis = millis();

  // 1. Post Sensor Data
  if (currentMillis - lastReadingTime >= readingInterval) {
    lastReadingTime = currentMillis;
    postSensorData();
  }

  // 2. Poll for Commands
  if (currentMillis - lastPollTime >= pollInterval) {
    lastPollTime = currentMillis;
    pollCommands();
  }
  
  // 3. Auto Mode Logic
  if (mode == "auto") {
    // Basic auto logic
    float moisture = readMoisture();
    if (moisture < moistureThreshold && !pumpStatus) {
      setPump(true);
    } else if (moisture >= (moistureThreshold + 10) && pumpStatus) {
      setPump(false);
    }
  }
}

float readMoisture() {
  int raw = analogRead(MOISTURE_PIN);
  // Map 0-4095 to 0-100% (calibrate these min/max values for your sensor)
  return map(raw, 4095, 0, 0, 100); 
}

void postSensorData() {
  if (WiFi.status() != WL_CONNECTED) return;

  float t = dht.readTemperature();
  float h = dht.readHumidity();
  float m = readMoisture();

  if (isnan(t) || isnan(h)) return;

  HTTPClient http;
  String url = String(supabaseUrl) + "/sensor_readings";
  http.begin(url);
  http.addHeader("Content-Type", "application/json");
  http.addHeader("apikey", supabaseKey);
  http.addHeader("Authorization", "Bearer " + String(supabaseKey));

  StaticJsonDocument<200> doc;
  doc["device_id"] = deviceId;
  doc["temperature"] = t;
  doc["humidity"] = h;
  doc["moisture"] = m;
  doc["pump_status"] = pumpStatus;
  doc["mode"] = mode;

  String payload;
  serializeJson(doc, payload);

  int httpCode = http.POST(payload);
  Serial.println("Post Data Code: " + String(httpCode));
  http.end();
}

void pollCommands() {
  if (WiFi.status() != WL_CONNECTED) return;

  HTTPClient http;
  String url = String(supabaseUrl) + "/device_commands?device_id=eq." + deviceId + "&executed=eq.false&limit=1";
  http.begin(url);
  http.addHeader("apikey", supabaseKey);
  http.addHeader("Authorization", "Bearer " + String(supabaseKey));

  int httpCode = http.GET();
  if (httpCode == 200) {
    String payload = http.getString();
    StaticJsonDocument<512> doc;
    DeserializationError error = deserializeJson(doc, payload);
    
    if (!error && doc.size() > 0) {
      JsonObject cmdObj = doc[0];
      String cmd = cmdObj["command"].as<String>();
      long cmdId = cmdObj["id"].as<long>();
      
      executeCommand(cmd);
      markCommandExecuted(cmdId);
    }
  }
  http.end();
}

void executeCommand(String cmd) {
  if (cmd == "pump_on") {
    setPump(true);
  } else if (cmd == "pump_off") {
    setPump(false);
  } else if (cmd == "mode_auto") {
    mode = "auto";
  } else if (cmd == "mode_manual") {
    mode = "manual";
  }
}

void setPump(bool state) {
  pumpStatus = state;
  digitalWrite(RELAY_PIN, state ? HIGH : LOW);
}

void markCommandExecuted(long cmdId) {
  HTTPClient http;
  String url = String(supabaseUrl) + "/device_commands?id=eq." + String(cmdId);
  http.begin(url);
  http.addHeader("Content-Type", "application/json");
  http.addHeader("apikey", supabaseKey);
  http.addHeader("Authorization", "Bearer " + String(supabaseKey));

  String payload = "{\"executed\":true}";
  http.PATCH(payload);
  http.end();
}
```
